using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace ConsoleApplication1
{
    class Program
    {
        static void Main(string[] args)
        {

            int a, b, temp, num, i;
            a = 0; b = 1;
            i = 1;

            Console.Write("Enter range of febonacci series=");
            num = Convert.ToInt32(Console.ReadLine());

            Console.Write(a + " " + b);
            while (i < num)
            {

                temp = a + b;
                Console.Write(" "+temp );
                a = b;
                b = temp;
                i++;
            }
            Console.ReadLine();

        }
    }
}
